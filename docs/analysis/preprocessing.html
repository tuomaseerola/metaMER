hu2017models <- metaMER_results[str_detect(metaMER_results$citekey, 
                                           "hu2017cr"),]$unique_id

substr(hu2017models, 
       1, 
       nchar(hu2017models) - 1),

                  hu2017models,]$unique_id <- hu2017models_replacement
